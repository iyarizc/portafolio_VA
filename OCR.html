import cv2 #importamos pythoncv para manipular imagenes
import pytesseract #libreria ocr

pytesseract.pytesseract.tesseract_cmd=r"C:\Program Files\Tesseract-OCR\tesseract.exe"

#cargar imagen
imgtexto=cv2.imread("imagen 3.png")

#redimencionar el tama√±o de la imagen
imgfinal=cv2.resize(imgtexto,(1000,600))

#convertir a escala de grises
img_gris=cv2.cvtColor(imgfinal,cv2.COLOR_BGR2GRAY)

#separamos el fondo con el texto
img_res=cv2.threshold(img_gris,120,255,cv2.THRESH_BINARY)[1]
#devuelve 2 valores, los pixeles y la imagen procesada

#configurar OCR
texto=pytesseract.image_to_string(img_res,lang="spa")
#comom parametro le pasamos la variable con la imagen procesada y el idioma
#imprimimos el titulo
print("texto detectado")
print(texto)

cv2.imshow("imagen analizada",imgfinal)
cv2.waitKey(0)
cv2.destroyAllWindows()
